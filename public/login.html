async function handleLogin(event) { event.preventDefault(); try { const response = await fetch(`${API_URL}/auth/login`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email: document.getElementById('email').value, password: document.getElementById('password').value }) }); if (!response.ok) { throw new Error('Credenciais inv√°lidas'); } const data = await response.json(); console.log('Login response:', data); localStorage.setItem('token', data.token); localStorage.setItem('userId', data.user.id); console.log('Stored userId:', localStorage.getItem('userId')); window.location.href = '/dashboard.html'; } catch (error) { alert(error.message); } }
